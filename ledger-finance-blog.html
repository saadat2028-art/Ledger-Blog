compound-interest.js 
// Future value of a monthly contribution
function futureValue(principal, monthly, annualRate, years) {
  const r = annualRate / 12;
  const n = years * 12;
  let balance = principal;

  for (let i = 0; i < n; i++) {
    balance = balance * (1 + r) + monthly;
  }
  return Math.round(balance);
}

futureValue(1000, 200, 0.07, 20);
// -> 108,858
budget-split.js 
// 50/30/20 split of take-home pay
function splitBudget(takeHome) {
  return {
    needs: Math.round(takeHome * 0.50),
    wants: Math.round(takeHome * 0.30),
    savings: Math.round(takeHome * 0.20),
  };
}

splitBudget(4200);
// -> { needs: 2100, wants: 1260, savings: 840 }
